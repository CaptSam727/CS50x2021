CREATE TABLE users (
id INTEGER, 
username TEXT NOT NULL, 
hash TEXT NOT NULL, 
cash NUMERIC NOT NULL DEFAULT 10000.00, 
PRIMARY KEY(id));

CREATE TABLE stocks(
symbol TEXT NOT NULL,
name TEXT NOT NULL,
shares INT NOT NULL,
price FLOAT NOT NULL,
total FLOAT NOT NULL, 
id INTEGER,
operation TEXT,
time DATETIME CURRENT_TIMESTAMP);
